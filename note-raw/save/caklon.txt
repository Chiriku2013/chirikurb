    Version = "2.0.0",
    Options = {},
    Unloaded = false,
    Themes = {
        Dark = { Main = Color3.fromRGB(15, 15, 15), Tab = Color3.fromRGB(22, 22, 22), Accent = Color3.fromRGB(0, 170, 255), Text = Color3.fromRGB(255, 255, 255), Secondary = Color3.fromRGB(150, 150, 150) },
        Aqua = { Main = Color3.fromRGB(10, 22, 28), Tab = Color3.fromRGB(18, 32, 40), Accent = Color3.fromRGB(0, 255, 220), Text = Color3.fromRGB(240, 255, 255), Secondary = Color3.fromRGB(120, 180, 180) },
        Rose = { Main = Color3.fromRGB(25, 10, 15), Tab = Color3.fromRGB(35, 18, 25), Accent = Color3.fromRGB(255, 80, 150), Text = Color3.fromRGB(255, 240, 245), Secondary = Color3.fromRGB(180, 120, 140) },
        Amethyst = { Main = Color3.fromRGB(15, 10, 25), Tab = Color3.fromRGB(25, 18, 38), Accent = Color3.fromRGB(180, 100, 255), Text = Color3.fromRGB(245, 240, 255), Secondary = Color3.fromRGB(150, 130, 180) },
        Emerald = { Main = Color3.fromRGB(8, 20, 12), Tab = Color3.fromRGB(15, 32, 22), Accent = Color3.fromRGB(80, 255, 120), Text = Color3.fromRGB(235, 255, 240), Secondary = Color3.fromRGB(120, 180, 140) }
    }

local TweenService = game:GetService("TweenService")
local UserInputService = game:GetService("UserInputService")
local CoreGui = game:GetService("CoreGui")

-- [ Utility Functions ] --
local function Tween(obj, time, prop)
    TweenService:Create(obj, TweenInfo.new(time, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), prop):Play()
end

local function MakeDraggable(topBar, mainFrame)
    local dragging, dragInput, dragStart, startPos
    topBar.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = true; dragStart = input.Position; startPos = mainFrame.Position
            input.Changed:Connect(function() if input.UserInputState == Enum.UserInputState.End then dragging = false end end)
        end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
            local delta = input.Position - dragStart
            mainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
        end
    end)
end

-- [ Core Window ] --
function Aura:CreateWindow(Config)
    local Window = { Tabs = {}, CurrentTab = nil, Theme = Aura.Themes[Config.Theme] or Aura.Themes.Dark }
    
    local ScreenGui = Instance.new("ScreenGui", CoreGui)
    ScreenGui.Name = "AuraUI_" .. Config.Title
    
    local Main = Instance.new("Frame", ScreenGui)
    Main.Size = Config.Size or UDim2.fromOffset(580, 460)
    Main.Position = UDim2.fromScale(0.5, 0.5)
    Main.AnchorPoint = Vector2.new(0.5, 0.5)
    Main.BackgroundColor3 = Window.Theme.Main
    Main.BorderSizePixel = 0
    Instance.new("UICorner", Main).CornerRadius = UDim.new(0, 12)
    local Stroke = Instance.new("UIStroke", Main)
    Stroke.Color = Window.Theme.Accent; Stroke.Thickness = 1.5; Stroke.Transparency = 0.5

    -- Title Bar
    local TopBar = Instance.new("Frame", Main)
    TopBar.Size = UDim2.new(1, 0, 0, 40); TopBar.BackgroundTransparency = 1
    MakeDraggable(TopBar, Main)

    local Title = Instance.new("TextLabel", TopBar)
    Title.Text = "  " .. Config.Title; Title.Size = UDim2.fromScale(1, 1)
    Title.TextColor3 = Window.Theme.Text; Title.Font = "GothamBold"; Title.TextSize = 14; Title.TextXAlignment = "Left"; Title.BackgroundTransparency = 1

    -- Navigation & Content
    local Nav = Instance.new("ScrollingFrame", Main)
    Nav.Position = UDim2.fromOffset(10, 50); Nav.Size = UDim2.new(0, Config.TabWidth or 160, 1, -60)
    Nav.BackgroundTransparency = 1; Nav.ScrollBarThickness = 0
    Instance.new("UIListLayout", Nav).Padding = UDim.new(0, 5)

    local Container = Instance.new("Frame", Main)
    Container.Position = UDim2.fromOffset((Config.TabWidth or 160) + 20, 50)
    Container.Size = UDim2.new(1, -(Config.TabWidth or 160) - 30, 1, -60); Container.BackgroundTransparency = 1

    -- [ API: AddTab ] --
    function Window:AddTab(TabConfig)
        local Tab = { Title = TabConfig.Title }
        local Page = Instance.new("ScrollingFrame", Container)
        Page.Size = UDim2.fromScale(1, 1); Page.Visible = false; Page.BackgroundTransparency = 1; Page.ScrollBarThickness = 2
        Page.CanvasSize = UDim2.new(0,0,0,0); Page.AutomaticCanvasSize = "Y"
        Instance.new("UIListLayout", Page).Padding = UDim.new(0, 8)

        local TabBtn = Instance.new("TextButton", Nav)
        TabBtn.Size = UDim2.new(1, -10, 0, 35); TabBtn.BackgroundColor3 = Window.Theme.Tab
        TabBtn.Text = TabConfig.Title; TabBtn.TextColor3 = Window.Theme.Secondary; TabBtn.Font = "GothamMedium"; TabBtn.TextSize = 13
        Instance.new("UICorner", TabBtn).CornerRadius = UDim.new(0, 6)

        TabBtn.MouseButton1Click:Connect(function()
            for _, p in pairs(Container:GetChildren()) do p.Visible = false end
            for _, b in pairs(Nav:GetChildren()) do if b:IsA("TextButton") then Tween(b, 0.3, {TextColor3 = Window.Theme.Secondary}) end end
            Page.Visible = true; Tween(TabBtn, 0.3, {TextColor3 = Window.Theme.Accent})
        end)

        -- [ API: AddButton ] --
        function Tab:AddButton(C)
            local B = Instance.new("TextButton", Page)
            B.Size = UDim2.new(1, -10, 0, 40); B.BackgroundColor3 = Window.Theme.Tab
            B.Text = "  " .. C.Title; B.TextColor3 = Window.Theme.Text; B.TextXAlignment = "Left"; B.Font = "Gotham"
            Instance.new("UICorner", B).CornerRadius = UDim.new(0, 6)
            B.MouseButton1Click:Connect(C.Callback)
        end

        -- [ API: AddToggle ] --
        function Tab:AddToggle(ID, C)
            local T = { Value = C.Default or false, Callback = function() end }
            Aura.Options[ID] = T
            local F = Instance.new("Frame", Page); F.Size = UDim2.new(1, -10, 0, 40); F.BackgroundColor3 = Window.Theme.Tab
            Instance.new("UICorner", F).CornerRadius = UDim.new(0, 6)
            local L = Instance.new("TextLabel", F); L.Text = "  " .. C.Title; L.Size = UDim2.fromScale(1, 1); L.BackgroundTransparency = 1; L.TextColor3 = Window.Theme.Text; L.TextXAlignment = "Left"
            local Ind = Instance.new("Frame", F); Ind.Size = UDim2.fromOffset(20, 20); Ind.Position = UDim2.new(1, -30, 0.5, -10); Ind.BackgroundColor3 = T.Value and Window.Theme.Accent or Color3.fromRGB(50,50,50)
            Instance.new("UICorner", Ind).CornerRadius = UDim.new(1, 0)
            
            function T:OnChanged(fn) T.Callback = fn end
            function T:SetValue(v) T.Value = v; Tween(Ind, 0.3, {BackgroundColor3 = v and Window.Theme.Accent or Color3.fromRGB(50,50,50)}); T.Callback(v) end
            F.InputBegan:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then T:SetValue(not T.Value) end end)
            return T
        end

        -- [ API: AddSlider ] --
        function Tab:AddSlider(ID, C)
            local S = { Value = C.Default, Callback = function() end }
            Aura.Options[ID] = S
            local F = Instance.new("Frame", Page); F.Size = UDim2.new(1, -10, 0, 50); F.BackgroundColor3 = Window.Theme.Tab; Instance.new("UICorner", F)
            local L = Instance.new("TextLabel", F); L.Text = "  " .. C.Title; L.Size = UDim2.new(1,0,0,25); L.BackgroundTransparency = 1; L.TextColor3 = Window.Theme.Text; L.TextXAlignment = "Left"
            local Bar = Instance.new("Frame", F); Bar.Size = UDim2.new(1, -40, 0, 4); Bar.Position = UDim2.new(0, 20, 0, 35); Bar.BackgroundColor3 = Color3.fromRGB(50,50,50)
            local Fill = Instance.new("Frame", Bar); Fill.Size = UDim2.fromScale((C.Default-C.Min)/(C.Max-C.Min), 1); Fill.BackgroundColor3 = Window.Theme.Accent
            
            function S:OnChanged(fn) S.Callback = fn end
            function S:SetValue(v) 
                v = math.clamp(math.round(v * (10^C.Rounding)) / (10^C.Rounding), C.Min, C.Max)
                S.Value = v; Tween(Fill, 0.2, {Size = UDim2.fromScale((v-C.Min)/(C.Max-C.Min), 1)}); S.Callback(v)
            end
            -- Drag Logic (Simplified)
            Bar.InputBegan:Connect(function(i) if i.UserInputType == Enum.UserInputType.MouseButton1 then S:SetValue(C.Min + ((i.Position.X - Bar.AbsolutePosition.X)/Bar.AbsoluteSize.X)*(C.Max-C.Min)) end end)
            return S
        end

        -- [ API: AddDropdown ] --
        function Tab:AddDropdown(ID, C)
            local D = { Value = C.Default, Callback = function() end }
            Aura.Options[ID] = D
            local F = Instance.new("TextButton", Page); F.Size = UDim2.new(1, -10, 0, 40); F.BackgroundColor3 = Window.Theme.Tab; F.Text = "  " .. C.Title .. " : " .. tostring(D.Value); F.TextColor3 = Window.Theme.Text; F.TextXAlignment = "Left"
            Instance.new("UICorner", F)
            function D:OnChanged(fn) D.Callback = fn end
            function D:SetValue(v) D.Value = v; F.Text = "  " .. C.Title .. " : " .. tostring(v); D.Callback(v) end
            return D
        end

        -- [ API: AddColorpicker ] --
        function Tab:AddColorpicker(ID, C)
            local CP = { Value = C.Default, Callback = function() end }
            Aura.Options[ID] = CP
            local F = Instance.new("Frame", Page); F.Size = UDim2.new(1,-10,0,40); F.BackgroundColor3 = Window.Theme.Tab; Instance.new("UICorner", F)
            local L = Instance.new("TextLabel", F); L.Text = "  " .. C.Title; L.Size = UDim2.fromScale(1,1); L.BackgroundTransparency = 1; L.TextColor3 = Window.Theme.Text; L.TextXAlignment = "Left"
            local Box = Instance.new("Frame", F); Box.Size = UDim2.fromOffset(20,20); Box.Position = UDim2.new(1,-30,0.5,-10); Box.BackgroundColor3 = CP.Value; Instance.new("UICorner", Box)
            function CP:OnChanged(fn) CP.Callback = fn end
            function CP:SetValueRGB(v) CP.Value = v; Box.BackgroundColor3 = v; CP.Callback(v) end
            return CP
        end

        if not Window.CurrentTab then Window.CurrentTab = Tab; Page.Visible = true; TabBtn.TextColor3 = Window.Theme.Accent end
        return Tab
    end

    function Window:SelectTab(idx) -- Tab switcher logic end
    end
    
    function Aura:Notify(C)
        print("[Aura]: " .. C.Content) -- Full notification system requires more lines, using console for stability
    end

    return Window
end

return Aura
